from plotGraphs import * from js import document from pyodide import create_proxy import json import plotly import js def plot_graph(feature): g = Graphs(variable_name=feature) type = document.getElementById("type-dropdown").value if(type == "line-graph"): graph = g.line_plot() elif(type == "bar-graph"): graph = g.bar_plot() elif(type == "scatter-plot"): graph = g.scatter_plot() else: graph = g.sunburst_plot() graphJSON = json.dumps(graph, cls=plotly.utils.PlotlyJSONEncoder) js.plot(graphJSON,"graph") def click(event): choice = document.getElementById("property-dropdown").value print(choice) plot_graph(choice) def setup(): proxy = create_proxy(click) e = document.getElementById("go") e.addEventListener("click", proxy) setup()
